  var Min : Integer
  var Max : Integer

  method MaxInt(a: Array[Integer]) : Integer is
   var max : Integer.Min
   var i : Integer.get(1)
   while i.Less(a.Length) loop
   if a.get(i).Greater(max) then max := a.get(i) end
